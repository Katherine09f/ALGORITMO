Simulador de Función verificarPrimo

¿Cuál será el resultado de ejecutar el siguiente código?

function verificarPrimo(num) {
    if (num <= 1) return false;
    for (let i = 2; i < num; i++) {
        if (num % i === 0) return false;
    }
    return true;
}
console.log(verificarPrimo(7));
a. true
b. false
c. undefined
d. Error de ejecución
